# 字符串切片

// 直接字面量使用
let str: &str = "Hello World";

let word: String = String::from("Hello World");

let len = word.len();

let hello_str = &word[0..4];
let world_str = &word[6..len];
let hello_world_str = &word[..];

// 字符串转切片
